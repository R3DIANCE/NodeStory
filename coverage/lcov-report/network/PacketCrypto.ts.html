export function generateHeader(
  iv: Uint8Array,
  length: number,
  version: number,
): Buffer {
  let a = iv[2] | (iv[3] << 8);
 
  a ^= -(version + 1);
  const b = a ^ length;
 
  const header = Buffer.from([
    a & 0xff,
    (a >>> 8) & 0xff,
    b & 0xff,
    (b >>> 8) & 0xff,
  ]);
  return header;
}
 
export function encryptData(data: Buffer, iv: Uint8Array): Buffer {
  const encryptedData = cryptoConfig.enableShandaEncryption
    ? ShandaEncryption.encrypt(data)
    : data;
 
  return AESEncryption.transform(encryptedData, iv);
}
 
export function decryptData(data: Buffer, iv: Uint8Array): Buffer {
  const transformedData = AESEncryption.transform(data, iv);
 
  return cryptoConfig.enableShandaEncryption
    ? ShandaEncryption.decrypt(transformedData)
    : transformedData;
}
 
export function morphIV(iv: Uint8Array): Uint8Array {
  const newSequence = new Uint8Array([0xf2, 0x53, 0x50, 0xc6]);
 
  for (let i = 0; i < 4; i++) {
    const input = iv[i];
    const tableInput = ivShiftingKey[input];
    newSequence[0] += ivShiftingKey[newSequence[1]] - input;
    newSequence[1] -= newSequence[2] ^ tableInput;
    newSequence[2] ^= ivShiftingKey[newSequence[3]] + input;
    newSequence[3] -= newSequence[0] - tableInput;
 
    let val =
      (newSequence[0] |
        ((newSequence[1] & 0xff) << 8) |
        ((newSequence[2] & 0xff) << 16) |
        ((newSequence[3] & 0xff) << 24)) >>>
      0;
    let val2 = val >>> 0x1d;
    val = (val << 0x03) >>> 0;
    val2 |= val;
    newSequence[0] = val2 & 0xff;
    newSequence[1] = (val2 >> 8) & 0xff;

    newSequence[3] = (val2 >> 24) & 0xff;
  }
 
  return newSequence;
}
